# GigBPM Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build GigBPM — a mobile-first PWA that runs a live metronome across a musician's entire setlist during performance.

**Architecture:** Single-page Svelte 5 app with no router library — a reactive `screen` variable in `App.svelte` switches between four screens. All business logic lives in framework-independent TypeScript modules (`metronome.ts`, `storage.ts`, `tts.ts`, `midi.ts`, `importexport.ts`) consumed by Svelte stores. The metronome engine uses Web Audio API lookahead scheduling exclusively — individual click timestamps are derived from `AudioContext.currentTime`; `setTimeout` drives only the scheduler loop cadence (not individual click timing).

**Tech Stack:** Svelte 5, Vite, TypeScript, vite-plugin-pwa (Workbox cache-first), Web Audio API, Web Speech API, Web MIDI API, `svelte-dnd-action` (drag-to-reorder), Vitest for unit tests

```ts
import type { Setlist, AppSettings } from './types'
import { DEFAULT_SETTINGS } from './types'

const SETLISTS_KEY = 'gigbpm_setlists'
const SETTINGS_KEY = 'gigbpm_settings'

export function loadSetlists(): Setlist[] {
  try {
    const raw = localStorage.getItem(SETLISTS_KEY)
    return raw ? JSON.parse(raw) : []
  } catch {
    return []
  }
}

export function saveSetlists(setlists: Setlist[]): void {
  localStorage.setItem(SETLISTS_KEY, JSON.stringify(setlists))
}

export function loadSettings(): AppSettings {
  try {
    const raw = localStorage.getItem(SETTINGS_KEY)
    if (!raw) return { ...DEFAULT_SETTINGS, midi: { ...DEFAULT_SETTINGS.midi } }
    const stored = JSON.parse(raw)
    return {
      ...DEFAULT_SETTINGS,
      ...stored,
      midi: { ...DEFAULT_SETTINGS.midi, ...stored.midi },
    }
  } catch {
    return { ...DEFAULT_SETTINGS, midi: { ...DEFAULT_SETTINGS.midi } }
  }
}

export function saveSettings(settings: AppSettings): void {
  localStorage.setItem(SETTINGS_KEY, JSON.stringify(settings))
}
```

```ts
// Timing accuracy is the single most critical requirement of this module.
// Individual click timestamps use AudioContext.currentTime exclusively.
// setTimeout below drives only the scheduler loop cadence — not click timing.

const LOOKAHEAD = 0.1        // seconds: how far ahead to schedule clicks
const SCHEDULE_INTERVAL = 25 // ms: how often to run the scheduler loop

export interface Metronome {
  start(bpm: number): void
  stop(): void
  pause(): void
  resume(): void
  setBpm(bpm: number): void
  onBeat(callback: () => void): void  // replaces previous; one listener at a time
  readonly isRunning: boolean
  readonly isPaused: boolean
}

export function createMetronome(ctx: AudioContext): Metronome {
  let bpm = 120
  let nextBeatTime = 0
  let schedulerTimer: ReturnType<typeof setTimeout> | null = null
  let beatCallback: (() => void) | null = null
  let running = false
  let paused = false
  let clickBuffer: AudioBuffer | null = null

  function buildClickBuffer(): AudioBuffer {
    const length = Math.floor(ctx.sampleRate * 0.04) // 40ms woodblock
    const buf = ctx.createBuffer(1, length, ctx.sampleRate)
    const data = buf.getChannelData(0)
    for (let i = 0; i < length; i++) {
      const t = i / ctx.sampleRate
      data[i] = (Math.random() * 2 - 1) * Math.exp(-t * 150)
    }
    return buf
  }

  function scheduleClick(time: number): void {
    if (!clickBuffer) clickBuffer = buildClickBuffer()
    const src = ctx.createBufferSource()
    src.buffer = clickBuffer
    src.connect(ctx.destination)
    src.start(time) // time from AudioContext.currentTime — not setTimeout

    // Fire beat callback at the scheduled moment for visual sync
    const delayMs = Math.max(0, (time - ctx.currentTime) * 1000)
    setTimeout(() => beatCallback?.(), delayMs)
  }

  function schedulerLoop(): void {
    // This setTimeout drives loop cadence only — not individual click timing
    while (nextBeatTime < ctx.currentTime + LOOKAHEAD) {
      scheduleClick(nextBeatTime)
      nextBeatTime += 60 / bpm
    }
    schedulerTimer = setTimeout(schedulerLoop, SCHEDULE_INTERVAL)
  }

  function stopLoop(): void {
    if (schedulerTimer !== null) {
      clearTimeout(schedulerTimer)
      schedulerTimer = null
    }
  }

  return {
    get isRunning() { return running },
    get isPaused() { return paused },

    start(newBpm: number): void {
      if (ctx.state === 'suspended') ctx.resume()
      bpm = newBpm
      nextBeatTime = ctx.currentTime
      running = true
      paused = false
      schedulerLoop()
    },

    stop(): void {
      stopLoop()
      running = false
      paused = false
    },

    pause(): void {
      stopLoop()
      running = false
      paused = true
    },

    resume(): void {
      if (!paused) return
      nextBeatTime = ctx.currentTime
      running = true
      paused = false
      schedulerLoop()
    },

    setBpm(newBpm: number): void {
      bpm = newBpm
      // Next scheduled beat will use the new BPM automatically
    },

    onBeat(callback: () => void): void {
      beatCallback = callback
    },
  }
}
```

```ts
export function isTTSAvailable(): boolean {
  return typeof window !== 'undefined' && 'speechSynthesis' in window
}

export function announce(text: string): void {
  if (!isTTSAvailable()) return
  window.speechSynthesis.cancel()
  const utterance = new SpeechSynthesisUtterance(text)
  window.speechSynthesis.speak(utterance)
}
```

```ts
import type { MidiCCBinding } from './types'

export function isMidiAvailable(): boolean {
  return typeof navigator !== 'undefined' && 'requestMIDIAccess' in navigator
}

export function matchesBinding(
  incoming: { channel: number; cc: number },
  binding: MidiCCBinding
): boolean {
  const channelOk = binding.channel === 'any' || binding.channel === incoming.channel
  return channelOk && binding.cc === incoming.cc
}

export interface MidiController {
  enable(): Promise<void>
  disable(): void
  setBindings(advance: MidiCCBinding | null, pauseStop: MidiCCBinding | null): void
  startLearn(target: 'advance' | 'pauseStop', onLearned: (b: MidiCCBinding) => void): void
  cancelLearn(): void
  simulateCC(channel: number, cc: number, value: number): void // test helper
}

export function createMidiController(
  onAdvance: () => void,
  onPauseStop: () => void
): MidiController {
  let advanceBinding: MidiCCBinding | null = null
  let pauseStopBinding: MidiCCBinding | null = null
  let learnTarget: 'advance' | 'pauseStop' | null = null
  let learnCallback: ((b: MidiCCBinding) => void) | null = null
  let midiAccess: MIDIAccess | null = null

  function handleCC(channel: number, cc: number, value: number): void {
    if (value === 0) return // ignore pedal release

    if (learnTarget && learnCallback) {
      learnCallback({ channel, cc })
      learnTarget = null
      learnCallback = null
      return
    }

    if (advanceBinding && matchesBinding({ channel, cc }, advanceBinding)) onAdvance()
    if (pauseStopBinding && matchesBinding({ channel, cc }, pauseStopBinding)) onPauseStop()
  }

  function onMidiMessage(e: MIDIMessageEvent): void {
    if (!e.data || e.data.length < 3) return
    const status = e.data[0]
    const cc = e.data[1]
    const value = e.data[2]
    const type = status & 0xf0
    const channel = (status & 0x0f) + 1
    if (type === 0xb0) handleCC(channel, cc, value) // CC message only
  }

  return {
    async enable(): Promise<void> {
      if (!isMidiAvailable()) return
      try {
        midiAccess = await navigator.requestMIDIAccess()
        const attach = () =>
          midiAccess!.inputs.forEach(input => { input.onmidimessage = onMidiMessage })
        attach()
        midiAccess.onstatechange = attach
      } catch { /* MIDI access denied — silently ignore */ }
    },

    disable(): void {
      if (midiAccess) {
        midiAccess.inputs.forEach(input => { input.onmidimessage = null })
        midiAccess = null
      }
    },

    setBindings(advance, pauseStop): void {
      advanceBinding = advance
      pauseStopBinding = pauseStop
    },

    startLearn(target, onLearned): void {
      learnTarget = target
      learnCallback = onLearned
    },

    cancelLearn(): void {
      learnTarget = null
      learnCallback = null
    },

    simulateCC(channel, cc, value): void {
      handleCC(channel, cc, value)
    },
  }
}
```

```ts
// AudioContext is constructed here (starts suspended on iOS — resumed in enter() via metronome.start())
const audioCtx = new AudioContext()
const metronome: Metronome = createMetronome(audioCtx)

// Live MIDI controller for performance — separate from the learn-only instance in Settings
const midiCtrl: MidiController = createMidiController(
  () => performanceStore.next(),       // advance binding fires Next
  () => {                               // pauseStop binding toggles pause/resume
    if (performanceStore.running) performanceStore.pause()
    else performanceStore.resume()
  }
)

function createPerformanceStore() {
  let setlist = $state<Setlist | null>(null)
  let songIndex = $state(0)
  let running = $state(false)
  let paused = $state(false)

  function current() {
    return setlist?.songs[songIndex] ?? null
  }

  function maybeAnnounce(): void {
    if (settingsStore.announceSongName && isTTSAvailable()) {
      const song = current()
      if (song) announce(song.name)
    }
  }

  return {
    get setlist() { return setlist },
    get songIndex() { return songIndex },
    get running() { return running },
    get paused() { return paused },
    get currentSong() { return current() },
    get totalSongs() { return setlist?.songs.length ?? 0 },
    get metronome() { return metronome },

    enter(sl: Setlist): void {
      setlist = sl
      songIndex = 0
      const song = current()
      if (!song) return
      running = true
      paused = false
      metronome.start(song.bpm)
      maybeAnnounce()
      // Wire MIDI bindings from settings for live performance
      if (settingsStore.midi.enabled) {
        midiCtrl.setBindings(settingsStore.midi.advance, settingsStore.midi.pauseStop)
        midiCtrl.enable()
      }
    },

    exit(): void {
      metronome.stop()
      midiCtrl.disable()
      setlist = null
      songIndex = 0
      running = false
      paused = false
    },

    next(): void {
      if (!setlist) return
      songIndex = (songIndex + 1) % setlist.songs.length
      const song = current()
      if (!song) return
      metronome.setBpm(song.bpm)
      maybeAnnounce()
      if (!running && !paused) {
        running = true
        metronome.resume()
      }
    },

    pause(): void {
      metronome.pause()
      running = false
      paused = true
    },

    resume(): void {
      metronome.resume()
      running = true
      paused = false
    },
  }
}
```

```ts
import type { Setlist } from './types'

export interface ExportPayload {
  version: 1
  setlist: {
    name: string
    songs: Array<{ name: string; bpm: number }>
  }
}

export function validateImport(data: unknown): Setlist {
  if (!data || typeof data !== 'object') throw new Error('Not an object')
  const d = data as Record<string, unknown>
  const sl = d.setlist as Record<string, unknown>
  if (!sl || typeof sl !== 'object') throw new Error('Missing setlist')
  if (typeof sl.name !== 'string') throw new Error('Invalid name')
  if (!Array.isArray(sl.songs)) throw new Error('Invalid songs')

  const songs = (sl.songs as unknown[]).map((s: unknown) => {
    if (!s || typeof s !== 'object') throw new Error('Invalid song')
    const song = s as Record<string, unknown>
    if (typeof song.name !== 'string') throw new Error('Invalid song name')
    if (typeof song.bpm !== 'number') throw new Error('Invalid song bpm')
    return { id: crypto.randomUUID(), name: song.name, bpm: song.bpm }
  })

  return { id: crypto.randomUUID(), name: sl.name, songs }
}

export function buildExportPayload(setlist: Setlist): ExportPayload {
  return {
    version: 1,
    setlist: {
      name: setlist.name,
      songs: setlist.songs.map(({ name, bpm }) => ({ name, bpm })),
    },
  }
}

export function exportSetlist(setlist: Setlist): void {
  const json = JSON.stringify(buildExportPayload(setlist), null, 2)
  const blob = new Blob([json], { type: 'application/json' })
  const url = URL.createObjectURL(blob)
  const a = document.createElement('a')
  a.href = url
  a.download = `setlist-${setlist.name.toLowerCase().replace(/[^a-z0-9]+/g, '-')}.json`
  document.body.appendChild(a)
  a.click()
  document.body.removeChild(a)
  URL.revokeObjectURL(url)
}
```
